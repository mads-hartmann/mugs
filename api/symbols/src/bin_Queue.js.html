1 /**
  2   @fileoverview Contains the implementation of the Queue ADT based on two Lists as
  3                 described in Chris Okasaki's book Purely Functional Data Structures. <br />
  4 
  5   @author Mads Hartmann Jensen (mads379@gmail.com)
  6 */var List, Queue;
  7 var __slice = Array.prototype.slice;
  8 if (typeof require != "undefined" && require !== null) {
  9   List = require('./list');
 10 }
 11 /**
 12   Queue provides the implementation of the abstract data type Queue based on two Lists as
 13   described in Chris Okasaki's book Purely Functional Data Structures. The Queue contains
 14   the following operations:
 15 
 16   <pre>
 17   --------------------------------------------------------
 18   Core operations of the Queue ADT
 19   --------------------------------------------------------
 20   enqueue()                               (amortized) O(1)
 21   dequeue(elem)                                       O(1)
 22   top()                                               O(1)
 23   --------------------------------------------------------
 24   </pre>
 25 
 26   @class Queue provides an implementation of the Queue ADT based on two Lists.
 27   @public
 28 */
 29 Queue = function() {
 30   var elements, f, half, r, size;
 31   elements = 1 <= arguments.length ? __slice.call(arguments, 0) : [];
 32   size = elements.length;
 33   half = Math.ceil(size / 2);
 34   f = elements.slice(0, half);
 35   r = elements.slice(half, size);
 36   this.front__ = new List(f);
 37   this.rear__ = new List(r).reverse();
 38   return this;
 39 };
 40 /**
 41   Removes the front element from the queue
 42 
 43   @return A new queue without the former front element
 44 */
 45 Queue.prototype.dequeue = function() {
 46   return this.buildFromLists(this.front__.tail(), this.rear__);
 47 };
 48 /**
 49   Adds a new element to the queue
 50 
 51   @param elem The element to add to the queue
 52   @return A new queue with the element in the back of the queue
 53 */
 54 Queue.prototype.enqueue = function(elem) {
 55   return this.buildFromLists(this.front__, this.rear__.prepend(elem));
 56 };
 57 /**
 58   Reads the front element in the queue
 59 
 60   @return The front element in the queue
 61 */
 62 Queue.prototype.front = function() {
 63   return this.front__.head();
 64 };
 65 /**
 66   Returns a List with all of the elements in the queue.
 67 
 68   @return A List with all of the elements in the queue.
 69 */
 70 Queue.prototype.values = function() {
 71   return this.front__.appendList(this.rear__.reverse());
 72 };
 73 /**
 74   Given two lists this will create a new queue. This is the method that makes sure the
 75   invariant "front is empty only if rear is empty" is kept.
 76 
 77   @param front The list that's going to be the front of the queue
 78   @param rear The list that's going to be the rear of the queue
 79   @private
 80 */
 81 Queue.prototype.buildFromLists = function(front, rear) {
 82   var queue;
 83   queue = new Queue();
 84   if (front.isEmpty()) {
 85     queue.front__ = rear.reverse();
 86     queue.rear__ = front;
 87   } else {
 88     queue.rear__ = rear;
 89     queue.front__ = front;
 90   }
 91   return queue;
 92 };
