 11 /**
 12   Don't use the Option prototype directly, use the {@link Some} or {@link None}. An option represents
 13   an optional value. It consists of either Some or None.
 14 
 15   @see Some
 16   @see None
 17   @constructor
 18 */
 19 Option = function() {
 20   return this;
 21 };
 22 /**
 23   Checks if the Option is empty.
 24   @return {boolean} True if the Option is empty, i.e. of type None. Otherwise false.
 25 */
 26 Option.prototype.isEmpty = function() {
 27   throw new Error("Not implemented in Option");
 28 };
 29 /**
 30   Returns the option's value if the option is nonempty. This will throw an exception if
 31   invoked on an instance of None.
 32   @return {*} The value stored in the Option.
 33 */
 34 Option.prototype.get = function() {
 35   throw new Error("Not implemented in Option");
 36 };
 37 /**
 38   Returns the option's value if the option is nonempty, otherwise return the value
 39   'otherwise'.
 40   @param {*} otherwise The value to return if the Option is None.
 41   @return The option's value if Some otherwise the value passed to the function
 42 */
 43 Option.prototype.getOrElse = function(otherwise) {
 44   if (this.isEmpty()) {
 45     return otherwise;
 46   } else {
 47     return this.get();
 48   }
 49 };
 50 /**
 51   Returns a Some containing the result of applying f to this Option's
 52   value if this Option is nonempty. Otherwise return None.
 53 
 54   @param {function(*):*} f The unary function to apply
 55   @return {Option} Some containing the result of applying the function f on the
 56                    element inside the Option if nonempty. Otherwise None.
 57 */
 58 Option.prototype.map = function(f) {
 59   if (this.isEmpty()) {
 60     return new None;
 61   } else {
 62     return new Some(f(this.get()));
 63   }
 64 };
 65 /**
 66   Returns the result of applying f to this option's value if this Option is
 67   nonempty, otherwise None. This is Slightly different from `map` in that f is
 68   expected to return an Option (which could be None).
 69 
 70   @param {function(*): Option} f The unary function to apply
 71   @return {Option} The result of applying the function f on the
 72                    element inside the Option if nonempty. Otherwise None.
 73 */
 74 Option.prototype.flatMap = function(f) {
 75   if (this.isEmpty()) {
 76     return new None;
 77   } else {
 78     return f(this.get());
 79   }
 80 };
 81 /**
 82   Returns this Option if it is nonempty '''and''' applying the 'predicate' to
 83   this Option's value returns true. Otherwise, return None
 84 
 85   @param {function(*): boolean} predicate A function that decides if the value
 86                                 should be included in the resulting collection
 87 */
 88 Option.prototype.filter = function(predicate) {
 89   if (this.isEmpty() || predicate(this.get())) {
 90     return this;
 91   } else {
 92     return new None();
 93   }
 94 };
 95 /**
 96   Apply the given procedure f to the option's value, if it is nonempty. Otherwise,
 97   do nothing.
 98 
 99   @param {function(*): *} f A unary function.
100   @return Nothing. This is all about side effects
101 */
102 Option.prototype.foreach = function(f) {
103   if (!this.isEmpty()) {
104     return f(this.get());
105   }
106 };
107 /**
108   Some(x) represents an existing value x. It's part of the {@link Option} monad.
109 
110   @class Some(x) represents an existing value x. It's part of the {@link Option} monad.
111   @param {*} value The value to wrap inside an Option
112   @see Option
113   @constructor
114   @augments Option
115 */
116 Some = function(value) {
117   if (value === void 0) {
118     return new None();
119   } else {
120     this.__value = value;
121   }
122   return this;
123 };
124 Some.prototype = new Option();
125 Some.prototype.constructor = Some;
126 /**
127   See documentation in {@link Option}
128 */
129 Some.prototype.isEmpty = function() {
130   return false;
131 };
132 /**
133   See documentation in {@link Option}
134 */
135 Some.prototype.get = function() {
136   return this.__value;
137 };
138 /**
139   None represents the non-existing of a value. It's part of the {@link Option} monad.
140 
141   @see Option
142   @constructor
143   @augments Option
144   @class None represents the non-existing of a value. It's part of the {@link Option} monad.
145 */
146 None = function() {
147   return this;
148 };
149 None.prototype = new Option();
150 None.prototype.constructor = None;
151 /**
152   See documentation in {@link Option}
153 */
154 None.prototype.isEmpty = function() {
155   return true;
156 };
157 /**
158   See documentation in {@link Option}
159 */
160 None.prototype.get = function() {
161   throw new Error("Get get a value in an instance of None");
162 };
