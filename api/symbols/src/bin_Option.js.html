1 /**
  2 @fileoverview Contains the implementation of the Option data structure.
  3 @author Mads Hartmann Jensen (mads379@gmail.com)
  4 */var None, Some;
  5 Some = function(value) {
  6   if (value === void 0) {
  7     return new None();
  8   } else {
  9     this.__value = value;
 10   }
 11   this.isEmpty = function() {
 12     return false;
 13   };
 14   this.get = function() {
 15     return this.__value;
 16   };
 17   this.getOrElse = function(f) {
 18     return this.get();
 19   };
 20   return this;
 21 };
 22 None = function() {
 23   this.isEmpty = function() {
 24     return true;
 25   };
 26   this.get = function() {
 27     throw new Error("Can't get a value from None");
 28   };
 29   this.getOrElse = function(f) {
 30     return f();
 31   };
 32   return this;
 33 };
 34 if (typeof exports != "undefined" && exports !== null) {
 35   exports.Some = Some;
 36   exports.None = None;
 37 }
