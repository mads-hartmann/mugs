1 /**
  2   @fileoverview Contains the implementation of the Stack data structure based on a List <br />
  3 
  4   @author Mads Hartmann Jensen (mads379@gmail.com)
  5 */var List, Stack;
  6 var __slice = Array.prototype.slice;
  7 if (typeof require != "undefined" && require !== null) {
  8   List = require('./list');
  9 }
 10 /**
 11   Stack provides the implementation of the abstract data type Stack based on a List. The
 12   Stack contains the following operations:
 13 
 14   <pre>
 15   --------------------------------------------------------
 16   Core operations of the Stack ADT
 17   --------------------------------------------------------
 18   pop()                                               O(1)
 19   push( elem )                                        O(1)
 20   top()                                               O(1)
 21   --------------------------------------------------------
 22   </pre>
 23 
 24   @class Stack provides the implementation of the abstract data type Stack based on a List
 25   @param elements A repeatable argument with the elements you want on the Stack. The last
 26                   element will be on top of the stack.
 27   @public
 28 */
 29 Stack = function() {
 30   var elements;
 31   elements = 1 <= arguments.length ? __slice.call(arguments, 0) : [];
 32   this.list = new List(elements).reverse();
 33   return this;
 34 };
 35 /**
 36   Removes the top element from the stack.
 37 
 38   @return A new stack without the former top element
 39   @complexity O(1)
 40 */
 41 Stack.prototype.pop = function() {
 42   return this.buildFromList(this.list.tail());
 43 };
 44 /**
 45   Push an element on the stack
 46 
 47   @param elem the element to push on the stack
 48   @return A new stack with the new element on top
 49   @complexity O(1)
 50 */
 51 Stack.prototype.push = function(elem) {
 52   return this.buildFromList(this.list.prepend(elem));
 53 };
 54 /**
 55   Returns the top element of the stack.
 56 
 57   @return the top element.
 58   @complexity O(1)
 59 */
 60 Stack.prototype.top = function() {
 61   return this.list.head();
 62 };
 63 /**
 64   Returns a List with all of the elements in the stack
 65 
 66   @return A list with all of the elements in the Stack
 67   @complexity O(1)
 68 */
 69 Stack.prototype.values = function() {
 70   return this.list;
 71 };
 72 /**
 73   This will build a new Stack from a List. This is used internally.
 74 
 75   @complexity O(1)
 76   @private
 77 */
 78 Stack.prototype.buildFromList = function(list) {
 79   var stack;
 80   stack = new Stack();
 81   stack.list = list;
 82   return stack;
 83 };
 84 if (typeof exports != "undefined" && exports !== null) {
 85   exports.Stack = Stack;
 86 }
