  5 /**
  6   This is base prototype for all collections. It implements a set of methods for all collections
  7   in terms of the forEach method.
  8 
  9   This is to be considered a partial prototype. Prototypes which inherit from mahj.Traversable will have
 10   to implement two methods: forEach and buildFromArray
 11 
 12   The forEach method should be implemented as efficiently as possible as this is used to implement
 13   all the methods in the mahj.Traversable prototype. The buildFromArray method should simply be a constructor
 14   for the prototype which takes a plain old javascript array.
 15 */
 16 mahj.Traversable = function() {
 17   return this;
 18 };
 19 /**
 20   @private
 21 */
 22 mahj.Traversable.prototype.buildFromArray = function() {
 23   throw new Error("Should be implemented in subclass");
 24 };
 25 mahj.Traversable.prototype.forEach = function() {
 26   throw new Error("Should be implemented in subclass");
 27 };
 28 /**
 29   Returns a new collection with the values of applying the function 'f' on each element in 'this'
 30   collection.
 31 */
 32 mahj.Traversable.prototype.map = function(f) {
 33   var elements;
 34   elements = [];
 35   this.forEach(function(elem) {
 36     return elements.push(f(elem));
 37   });
 38   return new this.buildFromArray(elements);
 39 };
 40 /**
 41   Returns a new collection with the concatenated values of applying the function 'f' on each element
 42   in 'this' collection. The function 'f' is expected to return an object that implements the forEach
 43   method itself.
 44 */
 45 mahj.Traversable.prototype.flatMap = function(f) {
 46   var elements;
 47   elements = [];
 48   this.forEach(function(x) {
 49     return f(x).forEach(function(y) {
 50       return elements.push(y);
 51     });
 52   });
 53   return new this.buildFromArray(elements);
 54 };
 55 /**
 56 
 57 */
 58 mahj.Traversable.prototype.filter = function(f) {
 59   var elements;
 60   elements = [];
 61   this.forEach(function(elem) {
 62     if (f(elem)) {
 63       return elements.push(elem);
 64     }
 65   });
 66   return new this.buildFromArray(elements);
 67 };
 68 /**
 69 
 70 */
 71 mahj.Traversable.prototype.isEmpty = function() {
 72   var itIsEmpty;
 73   itIsEmpty = true;
 74   this.forEach(function(elem) {
 75     itIsEmpty = false;
 76     return false;
 77   });
 78   return true;
 79 };
