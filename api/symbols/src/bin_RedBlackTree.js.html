1 /**
  2   @fileoverview
  3 
  4   A Red-black tree is a binary tree with two invariants that ensure
  5   that the tree is balanced to keep the height of the tree low.
  6 
  7   1. No red node has a red child
  8   2. Every path from the root to an empty node contains
  9      the same number of black nodes
 10 
 11   The insert and contains methods have been implemented following Chris Okasaki's
 12   book 'purely functional data structures' and the remove operations has been
 13   implemented following this blog: http://matt.might.net/articles/red-black-delete/
 14 
 15   @author Mads Hartmann Jensen (mads379@gmail.com)
 16 */var BLACK, Cons, DOUBLE_BLACK, List, NEGATIVE_BLACK, Nil, None, Option, RED, RedBlackLeaf, RedBlackNode, Some;
 17 var __bind = function(fn, me){ return function(){ return fn.apply(me, arguments); }; };

 20   Some = Option.Some;
 21   None = Option.None;
 22   List = require('./list').List;
 23   Cons = require('./list').Cons;
 24   Nil = require('./list').Nil;
 25 }

 62 RedBlackLeaf = (function() {
 63   function RedBlackLeaf(color) {
 64     this.color = color;
 65     this.left = new None();
 66     this.right = new None();
 67     this.key = new None();
 68     this.value = new None();
 69   }
 70   RedBlackLeaf.prototype.isEmpty = function() {
 71     return true;
 72   };
 73   RedBlackLeaf.prototype.containsKey = function(key) {
 74     return false;
 75   };
 76   RedBlackLeaf.prototype.get = function(key) {
 77     return new None();
 78   };
 79   RedBlackLeaf.prototype.keys = function() {
 80     return new Nil();
 81   };
 82   RedBlackLeaf.prototype.values = function() {
 83     return new Nil();
 84   };
 85   RedBlackLeaf.prototype.inorderMap = function(func) {
 86     return new Nil();
 87   };

 90 RedBlackNode = (function() {
 91   var balance, bubble, standard_comparator;
 92   standard_comparator = function(elem1, elem2) {
 93     if (elem1 < elem2) {
 94       return -1;
 95     } else if (elem1 > elem2) {
 96       return 1;
 97     } else {
 98       return 0;
 99     }
100   };
101   RedBlackNode.color;
102   RedBlackNode.left;
103   RedBlackNode.key;
104   RedBlackNode.value;
105   RedBlackNode.right;
106   RedBlackNode.comparator;
107   function RedBlackNode(color, left, key, value, right, comparator) {
108     this.color = color;
109     this.left = left;
110     this.key = key;
111     this.value = value;
112     this.right = right;
113     this.comparator = comparator ? void 0 : standard_comparator;
114   }
115   RedBlackNode.prototype.get = function(key) {
116     var comp;
117     comp = this.comparator(key, this.key);
118     if (comp < 0) {
119       return this.left.get(key);
120     } else if (comp > 0) {
121       return this.right.get(key);
122     } else {
123       return new Some(this.value);
124     }
125   };
126   RedBlackNode.prototype.minKey = function() {
127     if (this.left.isEmpty()) {
128       return this.key;
129     } else {
130       return this.left.minKey();
131     }
132   };
133   RedBlackNode.prototype.maxKey = function() {
134     if (this.right.isEmpty()) {
135       return this.key;
136     } else {
137       return this.right.maxKey();
138     }
139   };
140   RedBlackNode.prototype.containsKey = function(key) {
141     var comp;
142     comp = this.comparator(key, this.key);
143     if (comp < 0) {
144       return this.left.containsKey(key);
145     } else if (comp > 0) {
146       return this.right.containsKey(key);
147     } else {
148       return true;
149     }
150   };
151   RedBlackNode.prototype.keys = function() {
152     return this.inorderMap(function(key, value) {
153       return key;
154     });
155   };
156   RedBlackNode.prototype.values = function() {
157     return this.inorderMap(function(key, value) {
158       return value;
159     });
160   };
161   RedBlackNode.prototype.inorderMap = function(f) {
162     var bigger, smaller, that;
163     smaller = !this.left.isEmpty() ? this.left.inorderMap(f) : new Nil();
164     bigger = !this.right.isEmpty() ? this.right.inorderMap(f) : new Nil();
165     that = new List(f(this.key, this.value));
166     return smaller.append(that).append(bigger);
167   };
168   RedBlackNode.prototype.insert = function(key, value) {
169     var t, __insert;
170     __insert = __bind(function(tree) {
171       var c, compare, k, l, r, v;
172       c = tree.color, l = tree.left, k = tree.key, v = tree.value, r = tree.right;
173       compare = this.comparator(key, k);
174       if (tree.isEmpty()) {
175         return new RedBlackNode(RED, new RedBlackLeaf(BLACK), key, value, new RedBlackLeaf(BLACK));
176       } else if (compare < 0) {
177         return balance(c, __insert(l), k, v, r);
178       } else if (compare > 0) {
179         return balance(c, l, k, v, __insert(r));
180       } else {
181         return tree;
182       }
183     }, this);
184     t = __insert(this);
185     return new RedBlackNode(BLACK, t.left, t.key, t.value, t.right);
186   };
187   RedBlackNode.prototype.remove = function(key) {
188     var compare, __rm;
189     __rm = __bind(function(tree) {
190       var isExternalNode, isInternalNode, isNodeWithOneChildLeft, isNodeWithOneChildRight, max, newLeftTree, value;
191       isExternalNode = function() {
192         return tree.left.isEmpty() && tree.right.isEmpty();
193       };
194       isInternalNode = function() {
195         return !tree.left.isEmpty() && !tree.right.isEmpty();
196       };
197       isNodeWithOneChildLeft = function() {
198         return (tree.color === BLACK) && (tree.left.color === RED && tree.right.isEmpty());
199       };
200       isNodeWithOneChildRight = function() {
201         return (tree.color === BLACK) && (tree.right.color === RED && tree.left.isEmpty());
202       };
203       if (tree.isEmpty()) {
204         return tree;
205       } else if (isExternalNode()) {
206         if (tree.color === RED) {
207           return new RedBlackLeaf(BLACK);
208         } else if (tree.color === BLACK) {
209           return new RedBlackLeaf(DOUBLE_BLACK);
210         }
211       } else if (isNodeWithOneChildLeft()) {
212         return new RedBlackNode(BLACK, tree.left.left, tree.left.key, tree.left.value, tree.left.right);
213       } else if (isNodeWithOneChildRight()) {
214         return new RedBlackNode(BLACK, tree.right.left, tree.right.key, tree.right.value, tree.right.right);
215       } else if (isInternalNode()) {
216         max = tree.left.maxKey();
217         value = tree.get(max).get;
218         newLeftTree = tree.left.remove(max);
219         return bubble(tree.color, newLeftTree, max, value, tree.right);
220       } else {
221         return new RedBlackNode(tree.color, tree.left, tree.key, tree.value, tree.right);
222       }
223     }, this);
224     compare = this.comparator(key, this.key);
225     if (compare < 0) {
226       return new RedBlackNode(this.color, this.left.remove(key), this.key, this.value, this.right);
227     } else if (compare > 0) {
228       return new RedBlackNode(this.color, this.left, this.key, this.value, this.right.remove(key));
229     } else {
230       return __rm(this);
231     }
232   };
233   bubble = function(color, left, key, value, right) {
234     if (left.color === DOUBLE_BLACK || right.color === DOUBLE_BLACK) {
235       return balance(color.add(), new RedBlackNode(left.color.subtract(), left.left, left.key, left.value, left.right), key, value, new RedBlackNode(right.color.subtract(), right.left, right.key, right.value, right.right));
236     } else {
237       return new RedBlackNode(color, left, key, value, right);
238     }
239   };
240   balance = function(color, left, key, value, right) {
241     var leftFollowedByLeft, leftFollowedByRight, leftIsNegativeBlack, rightFollowedByLeft, rightFollowedByRight, rightIsNegativeBlack;
242     leftFollowedByLeft = function() {
243       return (!left.isEmpty() && !left.left.isEmpty()) && (color === BLACK || color === DOUBLE_BLACK) && left.color === RED && left.left.color === RED;
244     };
245     leftFollowedByRight = function() {
246       return (!left.isEmpty() && !left.right.isEmpty()) && (color === BLACK || color === DOUBLE_BLACK) && left.color === RED && left.right.color === RED;
247     };
248     rightFollowedByLeft = function() {
249       return (!right.isEmpty() && !right.left.isEmpty()) && (color === BLACK || color === DOUBLE_BLACK) && right.color === RED && right.left.color === RED;
250     };
251     rightFollowedByRight = function() {
252       return (!right.isEmpty() && !right.right.isEmpty()) && (color === BLACK || color === DOUBLE_BLACK) && right.color === RED && right.right.color === RED;
253     };
254     leftIsNegativeBlack = function() {
255       return (!left.isEmpty()) && left.color === NEGATIVE_BLACK;
256     };
257     rightIsNegativeBlack = function() {
258       return (!right.isEmpty()) && right.color === NEGATIVE_BLACK;
259     };
260     if (leftFollowedByLeft()) {
261       return new RedBlackNode(color.subtract(), new RedBlackNode(BLACK, left.left.left, left.left.key, left.left.value, left.left.right), left.key, left.value, new RedBlackNode(BLACK, left.right, key, value, right));
262     } else if (leftFollowedByRight()) {
263       return new RedBlackNode(color.subtract(), new RedBlackNode(BLACK, left.left, left.key, left.value, left.right.left), left.right.key, left.right.value, new RedBlackNode(BLACK, right.right.right, key, value, right));
264     } else if (rightFollowedByLeft()) {
265       return new RedBlackNode(color.subtract(), new RedBlackNode(BLACK, left, key, value, right.left.left), right.left.key, right.left.value, new RedBlackNode(BLACK, right.left.right, right.key, right.value, right.right));
266     } else if (rightFollowedByRight()) {
267       return new RedBlackNode(color.subtract(), new RedBlackNode(BLACK, left, key, value, right.left), right.key, right.value, new RedBlackNode(BLACK, right.right.left, right.right.key, right.right.value, right.right.right));
268     } else if (leftIsNegativeBlack()) {
269       return new RedBlackNode(BLACK, new RedBlackNode(BLACK, balance(RED, left.left.left, left.left.key, left.left.value, left.left.right), left.key, left.value, left.right.left), left.right.key, left.right.value, new RedBlackNode(BLACK, right.left.right, key, value, right));
270     } else if (rightIsNegativeBlack()) {
271       return new RedBlackNode(BLACK, new RedBlackNode(BLACK, left, right.key, right.value, right.left.left), right.right.key, right.right.value, new RedBlackNode(BLACK, right.left.right, key, value, balance(RED, right.right.left, right.left.key, right.left.value, right.right.right)));
272     } else {
273       return new RedBlackNode(color, left, key, value, right);
274     }
275   };
276   RedBlackNode.prototype.isEmpty = function() {
277     return false;
278   };
279   return RedBlackNode;
280 })();
