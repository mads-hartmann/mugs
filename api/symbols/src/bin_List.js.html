 11 /**
 12   <p>List provides the implementation of the abstract data type 'List' based on a Singly-Linked list</p>
 13 
 14   @class List provides the implementation of the abstract data type 'List' based on a Singly-Linked list <br />
 15   @public
 16 */
 17 List = function() {
 18   var cons, elements, hd, tl, x, xs;
 19   elements = 1 <= arguments.length ? __slice.call(arguments, 0) : [];
 20   x = elements[0], xs = 2 <= elements.length ? __slice.call(elements, 1) : [];
 21   if (x === void 0 || (x instanceof Array && x.length === 0)) {
 22     return new Nil();
 23   }
 24   if (x instanceof Array) {
 25     hd = x[0], tl = 2 <= x.length ? __slice.call(x, 1) : [];
 26     this.head = hd;
 27     this.tail = new List(tl);
 28   } else {
 29     this.head = x;
 30     this.tail = new List(xs);
 31   }
 32   /**
 33     Helper method to construct a list from a value and another list
 34 
 35     @private
 36   */
 37   cons = function(head, tail) {
 38     var l;
 39     l = new List(head);
 40     l.tail = tail;
 41     return l;
 42   };
 43   this.isEmpty = function() {
 44     return false;
 45   };
 46   /**
 47     Update the value with index 'index'. This will copy all values up to the
 48     given index.
 49   */
 50   this.update = function(index, element) {
 51     if (index < 0) {
 52       throw new Error("Index out of bounds by " + index);
 53     } else if (index === 0) {
 54       return cons(element, this.tail);
 55     } else {
 56       return cons(this.head, this.tail.update(index - 1, element));
 57     }
 58   };
 59   /**
 60     Return an Option containt the nth element in the list. Some(element) if it
 61     exists. Otherwise None
 62   */
 63   this.get = function(index) {
 64     if (index < 0) {
 65       return new None();
 66     } else if (index === 0) {
 67       return new Some(this.head);
 68     } else {
 69       return this.tail.get(index - 1);
 70     }
 71   };
 72   /**
 73     Returns an Option containing the last element in the list
 74   */
 75   this.last = function() {
 76     if (this.tail.isEmpty()) {
 77       return new Some(this.head);
 78     } else {
 79       return this.tail.last();
 80     }
 81   };
 82   /**
 83     Returns an Option containing the first element in the list
 84   */
 85   this.first = function() {
 86     return new Some(this.head);
 87   };
 88   /**
 89     Create a new list by appending this value
 90   */
 91   this.append = function(element) {
 92     return cons(this.head, this.tail.append(element));
 93   };
 94   /**
 95     Create a new list by prepending this value
 96   */
 97   this.prepend = function(element) {
 98     return cons(element, this);
 99   };
100   /**
101     Creates a list by appending the argument list to 'this' list.
102 
103     Consider two lists. A is the argument passed to the method append
104     of list T. R is the resulting list. x2 is a copy of x1 etc.
105 
106     A: a -> b -> c -> Nil
107     T: x -> y -> z -> Nil
108     R: x2 -> y2 -> z2 -> A
109 
110     Thus all of the elements of A are reused.
111   */
112   this.appendList = function(list) {
113     return cons(this.head, this.tail.appendList(list));
114   };
115   /**
116     Creates a new list by copying all of the items in the argument 'list'
117     before of 'this' list
118 
119     Consider two lists. A is the argument passed to the method prepend
120     of list T. R is the resulting list. a2 is a copy of a1 etc.
121 
122     A: a -> b -> c -> Nil
123     T: x -> y -> z -> Nil
124     R: a2 -> b2 -> c3 -> T
125 
126     Thus all of the elements of T are reused
127   */
128   this.prependList = function(list) {
129     if (list.isEmpty()) {
130       return this;
131     } else {
132       return cons(list.head, this.prependList(list.tail));
133     }
134   };
135   /**
136     Applies a binary operator to a start value and all elements of this list,
137     going left to right
138   */
139   this.foldLeft = function(seed) {
140     return __bind(function(f) {
141       var __foldLeft;
142       __foldLeft = function(acc, xs) {
143         if (xs.isEmpty()) {
144           return acc;
145         } else {
146           return __foldLeft(f(acc, xs.head), xs.tail);
147         }
148       };
149       return __foldLeft(seed, this);
150     }, this);
151   };
152   /**
153     Applies a binary operator to a start value and all elements of this list,
154     going right to left
155   */
156   this.foldRight = function(seed) {
157     return __bind(function(f) {
158       var __foldRight;
159       __foldRight = function(xs) {
160         if (xs.isEmpty()) {
161           return seed;
162         } else {
163           return f(xs.head, __foldRight(xs.tail));
164         }
165       };
166       return __foldRight(this);
167     }, this);
168   };
169   return this;
170 };
171 Nil = function() {
172   this.head = new None();
173   this.tail = new None();
174   this.isEmpty = function() {
175     return true;
176   };
177   this.append = function(element) {
178     return new List(element);
179   };
180   this.prepend = function(element) {
181     return new List(element);
182   };
183   this.appendList = function(list) {
184     return list;
185   };
186   this.prependList = function(list) {
187     return list;
188   };
189   this.last = function() {
190     return new None();
191   };
192   this.first = function() {
193     return new None();
194   };
195   this.get = function(index) {
196     return new None();
197   };
198   this.update = function(index, element) {
199     throw new Error("Index out of bounds by " + index);
200   };
201   return this;
202 };
