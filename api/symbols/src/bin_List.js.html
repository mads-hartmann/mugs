 12 /**
 13   List provides the implementation of the abstract data type List based on a Singly-Linked list. The
 14   list contains the following operations:
 15 
 16   <pre>
 17   --------------------------------------------------------
 18   Core operations of the List ADT
 19   --------------------------------------------------------
 20   append( element )                                   O(n)
 21   prepend( element )                                  O(1)
 22   update( index, element )                            O(n)
 23   get( index )                                        O(n)
 24   remove( index )                                     O(n)
 25   --------------------------------------------------------
 26   Methods inherited from mahj.Traversable
 27   --------------------------------------------------------
 28   map( f )                                            O(n)
 29   flatMap( f )                                        O(n)
 30   filter( f )                                         O(n)
 31   forEach( f )                                        O(n)
 32   foldLeft(s)(f)                                      O(n)
 33   isEmpty()                                           O(1)
 34   contains( element )                                 O(n)    TODO
 35   forAll( f )                                         O(n)    TODO
 36   take( x )                                           O(n)    TODO
 37   takeWhile( f )                                      O(n)    TODO
 38   size()                                              O(n)    TODO
 39   --------------------------------------------------------
 40   </pre>
 41   @augments mahj.Traversable
 42   @class List provides the implementation of the abstract data type List based on a Singly-Linked list
 43   @public
 44 */
 45 List = function() {
 46   var elements, hd, tl, x, xs;
 47   elements = 1 <= arguments.length ? __slice.call(arguments, 0) : [];
 48   x = elements[0], xs = 2 <= elements.length ? __slice.call(elements, 1) : [];
 49   this.isEmpty = function() {
 50     return false;
 51   };
 52   if (x === void 0 || (x instanceof Array && x.length === 0)) {
 53     this.head = function() {
 54       throw new Error("Can't get head of empty List");
 55     };
 56     this.tail = function() {
 57       throw new Error("Can't get tail of empty List");
 58     };
 59     this.isEmpty = function() {
 60       return true;
 61     };
 62   } else if (x instanceof Array) {
 63     hd = x[0], tl = 2 <= x.length ? __slice.call(x, 1) : [];
 64     this.head = function() {
 65       return hd;
 66     };
 67     this.tail = function() {
 68       return new List(tl);
 69     };
 70   } else {
 71     this.head = function() {
 72       return x;
 73     };
 74     this.tail = function() {
 75       return new List(xs);
 76     };
 77   }
 78   return this;
 79 };
 80 List.prototype = new mahj.Traversable();
 81 List.prototype.constructor = List;

 87 /**
 88   Create a new list by appending this value
 89   @param {*} element The element to append to the List
 90   @return {List} A new list containing all the elements of the old with followed by the element
 91 */
 92 List.prototype.append = function(element) {
 93   if (this.isEmpty()) {
 94     return new List(element);
 95   } else {
 96     return this.cons(this.head(), this.tail().append(element));
 97   }
 98 };
 99 /**
100   Create a new list by prepending this value
101   @param {*} element The element to prepend to the List
102   @return {List} A new list containing all the elements of the old list prepended with the element
103 */
104 List.prototype.prepend = function(element) {
105   return this.cons(element, this);
106 };
107 /**
108   Update the value with the given index.
109   @param {number} index The index of the element to update
110   @param {*} element The element to replace with the current element
111   @return {List} A new list with the updated value.
112 */
113 List.prototype.update = function(index, element) {
114   if (index < 0) {
115     throw new Error("Index out of bounds by " + index);
116   } else if (index === 0) {
117     return this.cons(element, this.tail());
118   } else {
119     return this.cons(this.head(), this.tail().update(index - 1, element));
120   }
121 };
122 /**
123   Return an Option containing the nth element in the list.
124   @param {number} index The index of the element to get
125   @return {Some|None} Some(element) is it exists, otherwise None
126 */
127 List.prototype.get = function(index) {
128   if (index < 0 || this.isEmpty()) {
129     new None();
130     return new None();
131   } else if (index === 0) {
132     return new Some(this.head());
133   } else {
134     return this.tail().get(index - 1);
135   }
136 };
137 /**
138   Removes the element at the given index. Runs in O(n) time.
139   @param {number} index The index of the element to remove
140   @return {List} A new list without the element at the given index
141 */
142 List.prototype.remove = function(index) {
143   if (index === 0) {
144     if (!this.tail().isEmpty()) {
145       return this.cons(this.tail().first().get(), this.tail().tail);
146     } else {
147       return new List();
148     }
149   } else {
150     return this.cons(this.head(), this.tail().remove(index - 1));
151   }
152 };
153 /**
154   The last element in the list
155   @return {Some|None} Some(last) if it exists, otherwise None
156 */
157 List.prototype.last = function() {
158   if (this.tail().isEmpty()) {
159     return new Some(this.head());
160   } else {
161     return this.tail().last();
162   }
163 };
164 /**
165   The first element in the list
166   @return {Some|None} Some(first) if it exists, otherwise None
167 */
168 List.prototype.first = function() {
169   return new Some(this.head());
170 };
171 /**
172   Creates a list by appending the argument list to 'this' list.
173 
174   @example
175   new List(1,2,3).appendList(new List(4,5,6));
176   // returns a list with the element 1,2,3,4,5,6
177   @param {List} list The list to append to this list.
178   @return {List} A new list containing the elements of the appended List and the elements of the original List.
179 */
180 List.prototype.appendList = function(list) {
181   if (this.isEmpty()) {
182     return list;
183   } else {
184     return this.cons(this.head(), this.tail().appendList(list));
185   }
186 };
187 /**
188   Creates a new list by copying all of the items in the argument 'list'
189   before of 'this' list
190 
191   @example
192   new List(4,5,6).prependList(new List(1,2,3));
193   // returns a list with the element 1,2,3,4,5,6
194   @param {List} list The list to prepend to this list.
195   @return {List} A new list containing the elements of the prepended List and the elements of the original List.
196 */
197 List.prototype.prependList = function(list) {
198   if (this.isEmpty()) {
199     return list;
200   } else {
201     if (list.isEmpty()) {
202       return this;
203     } else {
204       return this.cons(list.head(), this.prependList(list.tail()));
205     }
206   }
207 };

219 /**
220   Applies function 'f' on each element in the list. This return nothing and is only invoked
221   for the side-effects of f.
222   @see mahj.Traversable
223 */
224 List.prototype.forEach = function(f) {
225   if (!this.isEmpty()) {
226     f(this.head());
227     return this.tail().forEach(f);
228   }
229 };

235 /**
236   Helper method to construct a list from a value and another list
237   @private
238 */
239 List.prototype.cons = function(head, tail) {
240   var l;
241   l = new List(head);
242   l.tail = function() {
243     return tail;
244   };
245   return l;
246 };
247 /**
248   Applies a binary operator on all elements of this list going left to right and ending with the
249   seed value. This is a curried function that takes a seed value which returns a function that
250   takes a function which will then be applied to the elements
251 
252   @example
253   new List(1,2,3,4,5).foldLeft(0)(function(acc,current){ acc+current })
254   // returns 15 (the sum of the elements in the list)
255 
256   @param {*} seed The value to use when the list is empty
257   @return {function(function(*, *):*):*} A function which takes a binary function
258 */
259 List.prototype.foldLeft = function(seed) {
260   return __bind(function(f) {
261     var __foldLeft;
262     __foldLeft = function(acc, xs) {
263       if (xs.isEmpty()) {
264         return acc;
265       } else {
266         return __foldLeft(f(acc, xs.head()), xs.tail());
267       }
268     };
269     return __foldLeft(seed, this);
270   }, this);
271 };
272 /**
273   Applies a binary operator on all elements of this list going right to left and ending with the
274   seed value. This is a curried function that takes a seed value which returns a function that
275   takes a function which will then be applied to the elements.
276 
277   @example
278   new List(1,2,3,4,5).foldRight(0)(function(acc,current){ acc+current })
279   // returns 15 (the sum of the elements in the list)
280 
281   @param {*} seed The value to use when the list is empty
282   @return {function(function(*, *):*):*} A function which takes a binary function
283 */
284 List.prototype.foldRight = function(seed) {
285   return __bind(function(f) {
286     var __foldRight;
287     __foldRight = function(xs) {
288       if (xs.isEmpty()) {
289         return seed;
290       } else {
291         return f(__foldRight(xs.tail()), xs.head());
292       }
293     };
294     return __foldRight(this);
295   }, this);
296 };
297 /**
298   Returns a new list with the elements in reversed order.
299   @return A new list with the elements in reversed order
300 */
301 List.prototype.reverse = function() {
302   var rest, result;
303   result = new List();
304   rest = this;
305   while (!rest.isEmpty()) {
306     result = result.prepend(rest.head());
307     rest = rest.tail();
308   }
309   return result;
310 };
