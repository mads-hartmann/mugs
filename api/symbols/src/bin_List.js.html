  2   @fileoverview Contains the implementation of the List abstract data type.
  3   @author Mads Hartmann Jensen (mads379@gmail.com)
  4 */var List, Nil, None, Option, Some;
  5 var __slice = Array.prototype.slice, __bind = function(fn, me){ return function(){ return fn.apply(me, arguments); }; };
  6 if (typeof require != "undefined" && require !== null) {
  7   Option = require('./option');
  8   Some = Option.Some;
  9   None = Option.None;
 10 }
 11 /**
 12   List provides the implementation of the abstract data type List based on a Singly-Linked list. The
 13   list contains the following operations:
 14 
 15   <pre>
 16   --------------------------------------------------------
 17   Core operations of the List ADT
 18   --------------------------------------------------------
 19   append( element )                                   O(n)
 20   prepend( element )                                  O(1)
 21   update( index, element )                            O(n)
 22   get( index )                                        O(n)
 23   remove( index )                                     O(n)
 24   --------------------------------------------------------
 25   Methods that all containers have to implement
 26   --------------------------------------------------------
 27   map( f )                                            O(n)    TODO
 28   flatMap( f )                                        O(n)    TODO
 29   filter( f )                                         O(n)    TODO
 30   forEach( f )                                        O(n)    TODO
 31   foldLeft(s)(f)                                      O(n)
 32   isEmpty()                                           O(1)    TODO
 33   contains( element )                                 O(n)    TODO
 34   forAll( f )                                         O(n)    TODO
 35   take( x )                                           O(n)    TODO
 36   takeWhile( f )                                      O(n)    TODO
 37   size()                                              O(n)    TODO
 38   --------------------------------------------------------
 39   </pre>
 40   @class List provides the implementation of the abstract data type List based on a Singly-Linked list
 41   @public
 42 */
 43 List = function() {
 44   var cons, elements, hd, tl, x, xs;
 45   elements = 1 <= arguments.length ? __slice.call(arguments, 0) : [];
 46   x = elements[0], xs = 2 <= elements.length ? __slice.call(elements, 1) : [];
 47   if (x === void 0 || (x instanceof Array && x.length === 0)) {
 48     return new Nil();
 49   }
 50   if (x instanceof Array) {
 51     hd = x[0], tl = 2 <= x.length ? __slice.call(x, 1) : [];
 52     this.head = hd;
 53     this.tail = new List(tl);
 54   } else {
 55     this.head = x;
 56     this.tail = new List(xs);
 57   }
 58   /**
 59     Helper method to construct a list from a value and another list
 60     @private
 61   */
 62   cons = function(head, tail) {
 63     var l;
 64     l = new List(head);
 65     l.tail = tail;
 66     return l;
 67   };
 68   /**
 69     Checks if the list is empty
 70     @return {boolean} true if container is empty, otherwise false.
 71   */
 72   this.isEmpty = function() {
 73     return false;
 74   };
 75   /**
 76     Create a new list by appending this value
 77     @param {*} element The element to append to the List
 78     @return {List} A new list containing all the elements of the old with followed by the element
 79   */
 80   this.append = function(element) {
 81     return cons(this.head, this.tail.append(element));
 82   };
 83   /**
 84     Create a new list by prepending this value
 85     @param {*} element The element to prepend to the List
 86     @return {List} A new list containing all the elements of the old list prepended with the element
 87   */
 88   this.prepend = function(element) {
 89     return cons(element, this);
 90   };
 91   /**
 92     Update the value with the given index.
 93     @param {number} index The index of the element to update
 94     @param {*} element The element to replace with the current element
 95     @return {List} A new list with the updated value.
 96   */
 97   this.update = function(index, element) {
 98     if (index < 0) {
 99       throw new Error("Index out of bounds by " + index);
100     } else if (index === 0) {
101       return cons(element, this.tail);
102     } else {
103       return cons(this.head, this.tail.update(index - 1, element));
104     }
105   };
106   /**
107     Return an Option containing the nth element in the list.
108     @param {number} index The index of the element to get
109     @return {Some|None} Some(element) is it exists, otherwise None
110   */
111   this.get = function(index) {
112     if (index < 0) {
113       return new None();
114     } else if (index === 0) {
115       return new Some(this.head);
116     } else {
117       return this.tail.get(index - 1);
118     }
119   };
120   /**
121     Removes the element at the given index. Runs in O(n) time.
122     @param {number} index The index of the element to remove
123     @return {List} A new list without the element at the given index
124   */
125   this.remove = function(index) {
126     if (index === 0) {
127       if (!this.tail.isEmpty()) {
128         return cons(this.tail.first().get(), this.tail.tail);
129       } else {
130         return new Nil;
131       }
132     } else {
133       return cons(this.head, this.tail.remove(index - 1));
134     }
135   };
136   /**
137     The last element in the list
138     @return {Some|None} Some(last) if it exists, otherwise None
139   */
140   this.last = function() {
141     if (this.tail.isEmpty()) {
142       return new Some(this.head);
143     } else {
144       return this.tail.last();
145     }
146   };
147   /**
148     The first element in the list
149     @return {Some|None} Some(first) if it exists, otherwise None
150   */
151   this.first = function() {
152     return new Some(this.head);
153   };
154   /**
155     Creates a list by appending the argument list to 'this' list.
156 
157     @example
158     new List(1,2,3).appendList(new List(4,5,6));
159     // returns a list with the element 1,2,3,4,5,6
160     @param {List} list The list to append to this list.
161     @return {List} A new list containing the elements of the appended List and the elements of the original List.
162   */
163   this.appendList = function(list) {
164     return cons(this.head, this.tail.appendList(list));
165   };
166   /**
167     Creates a new list by copying all of the items in the argument 'list'
168     before of 'this' list
169 
170     @example
171     new List(4,5,6).prependList(new List(1,2,3));
172     // returns a list with the element 1,2,3,4,5,6
173     @param {List} list The list to prepend to this list.
174     @return {List} A new list containing the elements of the prepended List and the elements of the original List.
175   */
176   this.prependList = function(list) {
177     if (list.isEmpty()) {
178       return this;
179     } else {
180       return cons(list.head, this.prependList(list.tail));
181     }
182   };
183   /**
184     Applies a binary operator on all elements of this list going left to right and ending with the
185     seed value. This is a curried function that takes a seed value which returns a function that
186     takes a function which will then be applied to the elements
187 
188     @example
189     new List(1,2,3,4,5).foldLeft(0)(function(acc,current){ acc+current })
190     // returns 15 (the sum of the elements in the list)
191 
192     @param {*} seed The value to use when the list is empty
193     @return {function(function(*, *):*):*} A function which takes a binary function
194   */
195   this.foldLeft = function(seed) {
196     return __bind(function(f) {
197       var __foldLeft;
198       __foldLeft = function(acc, xs) {
199         if (xs.isEmpty()) {
200           return acc;
201         } else {
202           return __foldLeft(f(acc, xs.head), xs.tail);
203         }
204       };
205       return __foldLeft(seed, this);
206     }, this);
207   };
208   /**
209     Applies a binary operator on all elements of this list going right to left and ending with the
210     seed value. This is a curried function that takes a seed value which returns a function that
211     takes a function which will then be applied to the elements.
212 
213     @example
214     new List(1,2,3,4,5).foldRight(0)(function(acc,current){ acc+current })
215     // returns 15 (the sum of the elements in the list)
216 
217     @param {*} seed The value to use when the list is empty
218     @return {function(function(*, *):*):*} A function which takes a binary function
219   */
220   this.foldRight = function(seed) {
221     return __bind(function(f) {
222       var __foldRight;
223       __foldRight = function(xs) {
224         if (xs.isEmpty()) {
225           return seed;
226         } else {
227           return f(xs.head, __foldRight(xs.tail));
228         }
229       };
230       return __foldRight(this);
231     }, this);
232   };
233   return this;
234 };
235 Nil = function() {
236   this.head = new None();
237   this.tail = new None();
238   this.isEmpty = function() {
239     return true;
240   };
241   this.append = function(element) {
242     return new List(element);
243   };
244   this.prepend = function(element) {
245     return new List(element);
246   };
247   this.appendList = function(list) {
248     return list;
249   };
250   this.prependList = function(list) {
251     return list;
252   };
253   this.last = function() {
254     return new None();
255   };
256   this.first = function() {
257     return new None();
258   };
259   this.get = function(index) {
260     return new None();
261   };
262   this.remove = function(index) {
263     return new Nil();
264   };
265   this.update = function(index, element) {
266     throw new Error("Index out of bounds by " + index);
267   };
268   return this;
269 };

271   exports.List = List;
272   exports.Nil = Nil;
