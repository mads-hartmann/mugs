 13 /**
 14   TreeMap provides the implementation of the abstract data type 'Map' based on a Red Black Tree. The
 15   map contains the following operations
 16 
 17   <pre>
 18   --------------------------------------------------------
 19   Core operations of the Map ADT
 20   --------------------------------------------------------
 21   insert( key,value )                               O(logn)
 22   get( index )                                      O(logn)
 23   remove( index )                                   O(logn)
 24   contains( key )                                   O(logn)
 25   --------------------------------------------------------
 26   Methods that all containers have to implement
 27   --------------------------------------------------------
 28   map( f )                                            O(n)
 29   flatMap( f )                                        O(n)
 30   filter( f )                                         O(n)
 31   forEach( f )                                        O(n)
 32   foldLeft(s)(f)                                      O(n)    TODO
 33   isEmpty()                                           O(1)    TODO
 34   contains( element )                                 O(n)    TODO
 35   forAll( f )                                         O(n)    TODO
 36   take( x )                                           O(n)    TODO
 37   takeWhile( f )                                      O(n)    TODO
 38   size()                                              O(n)    TODO
 39   --------------------------------------------------------
 40   </pre>
 41 
 42   @class TreeMap provides the implementation of the abstract data type 'Map' based on a Red Black Tree.
 43   @constructor
 44   @example
 45   var map = new TreeMap([
 46     {key: 1, value: "one"},
 47     {key: 4, value: "four"},
 48     {key: 3, value: "three"},
 49     {key: 2, value: "two"}
 50   ]);
 51   @param {Array} keyValuePairs An array containing objects with the properties key & value.
 52   @param {Function=} comparator A comparator function that can compare the keys (optional). Will use a
 53                                 default comparator for integers if no comparator is given
 54   @public
 55   @augments mahj.Traversable
 56 */
 57 TreeMap = function(keyValuePairs, comparator) {
 58   var kv, treeUnderConstruction, _i, _len;
 59   if (keyValuePairs instanceof Array && keyValuePairs.length > 0) {
 60     treeUnderConstruction = new RedBlackLeaf(BLACK);
 61     for (_i = 0, _len = keyValuePairs.length; _i < _len; _i++) {
 62       kv = keyValuePairs[_i];
 63       treeUnderConstruction = treeUnderConstruction.insert(kv.key, kv.value);
 64     }
 65     this.tree = treeUnderConstruction;
 66   } else {
 67     this.tree = new RedBlackLeaf(BLACK);
 68   }
 69   if (comparator != null) {
 70     this.tree.comparator = comparator;
 71   }
 72   return this;
 73 };
 74 TreeMap.prototype = new mahj.Traversable();
 75 /*
 76 ---------------------------------------------------------------------------------------------
 77 Methods related to the MAP ADT
 78 ---------------------------------------------------------------------------------------------
 79 */
 80 /**
 81   Return a new TreeMap containing the given (key,value) pair.
 82   @param {*} key The key to store the value by
 83   @param {*} value The value to store in the map
 84   @return {TreeMap} A new TreeMap that also contains the new key-value pair
 85 */
 86 TreeMap.prototype.insert = function(key, value) {
 87   return this.buildFromTree(this.tree.insert(key, value));
 88 };
 89 /**
 90   If a (key,value) pair exists return Some(value), otherwise None()
 91   @param {*} key The key of the value you want to read.
 92   @return {Some|None} Some(value) if it exists in the map. Otherwise None
 93 */
 94 TreeMap.prototype.get = function(key) {
 95   return this.tree.get(key);
 96 };
 97 /**
 98   Returns a new TreeMap without the given key-value pair.
 99   @param {*} key The key of the value you want to remove
100   @return {TreeMap} A new TreeMap that doesn't contain the key-value pair
101 */
102 TreeMap.prototype.remove = function(key) {
103   return this.buildFromTree(this.tree.remove(key));
104 };
105 /**
106   Returns a sorted list containing all of the keys in the TreeMap
107   @return {List} A sorted list containing all of the keys in the TreeMap
108 */
109 TreeMap.prototype.keys = function() {
110   return this.tree.keys();
111 };
112 /**
113   Returns a sorted list containing all of the values in the TreeMap
114   @return {List} sorted list containing all of the values in the TreeMap
115 */
116 TreeMap.prototype.values = function() {
117   return this.tree.values();
118 };
119 /**
120   Used to construct a TreeMap from RedBlackTree. This is intended
121   for internal use only. Would've marked it private if I could.
122   @private
123 */
124 TreeMap.prototype.buildFromTree = function(tree) {
125   var map;
126   map = new TreeMap(this.comparator);
127   map.tree = tree;
128   return map;
129 };
130 /*
131 ---------------------------------------------------------------------------------------------
132 Methods related to Traversable prototype
133 ---------------------------------------------------------------------------------------------
134 */
135 /**
136   @private
137 */
138 TreeMap.prototype.buildFromArray = function(arr) {
139   return new TreeMap(arr, this.comparator);
140 };
141 /**
142   Applies function 'f' on each value in the map. This return nothing and is only invoked
143   for the side-effects of f.
144   @see mahj.Traversable
145 */
146 TreeMap.prototype.forEach = function(f) {
147   return this.tree.inorderTraversal(f);
148 };
