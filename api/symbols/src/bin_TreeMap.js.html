  2   @fileoverview Contains the implementation of the Map data structure based on a Red-Black Tree
  3   @author Mads Hartmann Jensen (mads379@gmail.com)
  4 */var BLACK, RED, RedBlackLeaf, RedBlackNode, RedBlackNodeWrapper, TreeMap;
  5 if (typeof require != "undefined" && require !== null) {
  6   RedBlackNodeWrapper = require('../src/RedBlackTree');
  7   RedBlackNode = RedBlackNodeWrapper.RedBlackNode;
  8   RedBlackLeaf = RedBlackNodeWrapper.Leaf;
  9   RED = RedBlackNodeWrapper.RED;
 10   BLACK = RedBlackNodeWrapper.BLACK;

 12 /**
 13   TreeMap provides the implementation of the abstract data type 'Map' based on a Red Black Tree. The
 14   map contains the following operations
 15 
 16   <pre>
 17   --------------------------------------------------------
 18   Core operations of the Map ADT
 19   --------------------------------------------------------
 20   insert( key,value )                               O(logn)
 21   get( index )                                      O(logn)
 22   remove( index )                                   O(logn)
 23   contains( key )                                   O(logn)
 24   --------------------------------------------------------
 25   Methods that all containers have to implement
 26   --------------------------------------------------------
 27   map( f )                                            O(n)    TODO
 28   flatMap( f )                                        O(n)    TODO
 29   filter( f )                                         O(n)    TODO
 30   forEach( f )                                        O(n)    TODO
 31   foldLeft(s)(f)                                      O(n)    TODO
 32   isEmpty()                                           O(1)    TODO
 33   contains( element )                                 O(n)    TODO
 34   forAll( f )                                         O(n)    TODO
 35   take( x )                                           O(n)    TODO
 36   takeWhile( f )                                      O(n)    TODO
 37   size()                                              O(n)    TODO
 38   --------------------------------------------------------
 39   </pre>
 40 
 41    @class TreeMap provides the implementation of the abstract data type 'Map' based on a Red Black Tree.
 42    @public
 43 */
 44 TreeMap = function(key, value, comparator) {
 45   this.tree = new RedBlackNode(RED, new RedBlackLeaf(BLACK), key, value, new RedBlackLeaf(BLACK), comparator);
 46   if (comparator != null) {
 47     this.tree.comparator = comparator;
 48   }
 49   this.insert = function(key, value) {
 50     return this.constructFromTree(this.tree.insert(key, value));
 51   };
 52   this.get = function(key) {
 53     return this.tree.get(key);
 54   };
 55   this.remove = function(key) {
 56     return this.constructFromTree(this.tree.remove(key));
 57   };
 58   this.keys = function() {
 59     return this.tree.keys();
 60   };
 61   this.values = function() {
 62     return this.tree.values();
 63   };
 64   this.constructFromTree = function(tree) {
 65     var set;
 66     set = new TreeMap();
 67     set.tree = tree;
 68     return set;
 69   };
 70   return this;
 71 };
 72 if (typeof exports != "undefined" && exports !== null) {
 73   exports.TreeMap = TreeMap;
 74 }
