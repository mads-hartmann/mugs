  4 */var BLACK, RED, RedBlackLeaf, RedBlackNode, RedBlackNodeWrapper, TreeMap;
  5 if (typeof require != "undefined" && require !== null) {
  6   RedBlackNodeWrapper = require('../src/RedBlackTree');
  7   RedBlackNode = RedBlackNodeWrapper.RedBlackNode;
  8   RedBlackLeaf = RedBlackNodeWrapper.Leaf;

 12 /**
 13   <p>TreeMap provides the implementation of the abstract data type 'Map' based on a Red Black Tree.</p>
 14 
 15   <p>The insert and contains methods have been implemented following Chris Okasaki's book 'purely functional
 16   data structures' and the remove operations has been implemented following this blog:
 17   http://matt.might.net/articles/red-black-delete/</p>
 18 
 19   <p>The asymptotic running time for important operations are below:</p>
 20   <pre>
 21      Method                 big-O
 22    --------------------------------------------------------------------
 23    - contains               O(logn)
 24    - filter                 TODO
 25    - values                 O(n)
 26    - insert                 O(logn)
 27    - insertAll              TODO
 28    - map                    TODO
 29    - remove                 O(logn)
 30    - removeAll              TODO
 31    - get                    O(logn)
 32    </pre>
 33 
 34    @class TreeMap provides the implementation of the abstract data type 'Map' based on a Red Black Tree. <br />
 35    @public
 36 */
 37 TreeMap = function(key, value, comparator) {
 38   this.tree = new RedBlackNode(RED, new RedBlackLeaf(BLACK), key, value, new RedBlackLeaf(BLACK), comparator);
 39   if (comparator != null) {
 40     this.tree.comparator = comparator;
 41   }
 42   this.insert = function(key, value) {
 43     return this.constructFromTree(this.tree.insert(key, value));
 44   };
 45   this.get = function(key) {
 46     return this.tree.get(key);
 47   };
 48   this.remove = function(key) {
 49     return this.constructFromTree(this.tree.remove(key));
 50   };
 51   this.keys = function() {
 52     return this.tree.keys();
 53   };
 54   this.values = function() {
 55     return this.tree.values();
 56   };
 57   this.constructFromTree = function(tree) {
 58     var set;
 59     set = new TreeMap();
 60     set.tree = tree;
 61     return set;
 62   };
 63   return this;
 64 };
 65 if (typeof exports != "undefined" && exports !== null) {
 66   exports.TreeMap = TreeMap;
 67 }
