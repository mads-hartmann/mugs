  2   @fileoverview Contains the implementation of the Set data structure based on a Red-Black Tree
  3   @author Mads Hartmann Jensen (mads379@gmail.com)
  4 */var BLACK, RED, RedBlackLeaf, RedBlackNode, RedBlackNodeWrapper, Traversable, TreeSet;
  5 if (typeof require != "undefined" && require !== null) {
  6   RedBlackNodeWrapper = require('../src/RedBlackTree');
  7   RedBlackNode = RedBlackNodeWrapper.RedBlackNode;
  8   RedBlackLeaf = RedBlackNodeWrapper.Leaf;
  9   RED = RedBlackNodeWrapper.RED;
 10   BLACK = RedBlackNodeWrapper.BLACK;
 11   Traversable = (require('./Traversable')).Traversable;

 13 /**
 14   @augments mahj.Traversable
 15   @class TreeSet provides the implementation of the abstract data type Set based on a Red Black Tree
 16   @public
 17 */
 18 TreeSet = function(elements, comparator) {
 19   var element, treeUnderConstruction, _i, _len;
 20   if (elements instanceof Array && elements.length > 0) {
 21     treeUnderConstruction = new RedBlackLeaf(BLACK);
 22     for (_i = 0, _len = elements.length; _i < _len; _i++) {
 23       element = elements[_i];
 24       treeUnderConstruction = treeUnderConstruction.insert(element, element);
 25     }
 26     this.tree = treeUnderConstruction;
 27   } else {
 28     this.tree = new RedBlackLeaf(BLACK);
 29   }
 30   if (comparator != null) {
 31     this.tree.comparator = comparator;
 32   }
 33   return this;
 34 };
 35 TreeSet.prototype = new mahj.Traversable();
 36 /**
 37   Insert an element in the set. If the set already contains an element equal to the given value,
 38   it is replaced with the new value.
 39   @param element The element to insert into the set
 40 */
 41 TreeSet.prototype.insert = function(element) {
 42   return this.buildFromTree(this.tree.insert(element, element));
 43 };
 44 /**
 45   Delete an element from the set
 46   @param element The element to remove from the set
 47 */
 48 TreeSet.prototype.remove = function(element) {
 49   return this.buildFromTree(this.tree.remove(element));
 50 };
 51 /**
 52   Tests if the set contains the element
 53   @param element The element to check for
 54 */
 55 TreeSet.prototype.contains = function(element) {
 56   return this.tree.containsKey(element);
 57 };
 58 /**
 59   The elements of the set
 60   @return {List} A list containing all the element of the set
 61 */
 62 TreeSet.prototype.values = function() {
 63   return this.tree.values();
 64 };
 65 /**
 66   Used to construct a TreeMap from RedBlackTree. This is intended
 67   for internal use only. Would've marked it private if I could.
 68   @private
 69 */
 70 TreeSet.prototype.buildFromTree = function(tree) {
 71   var set;
 72   set = new TreeSet(this.comparator);
 73   set.tree = tree;
 74   return set;
 75 };
 76 /*
 77 ---------------------------------------------------------------------------------------------
 78 Methods related to Traversable prototype
 79 ---------------------------------------------------------------------------------------------
 80 */
 81 /**
 82   @private
 83 */
 84 TreeSet.prototype.buildFromArray = function(arr) {
 85   return new TreeSet(arr, this.comparator);
 86 };
 87 /**
 88   Applies function 'f' on each value in the map. This return nothing and is only invoked
 89   for the side-effects of f.
 90   @param f The unary function to apply on each element in the set.
 91   @see mahj.Traversable
 92 */
 93 TreeSet.prototype.forEach = function(f) {
 94   var q;
 95   q = function(kv) {
 96     return {
 97       key: f(kv.key),
 98       value: f(kv.value)
 99     };
100   };
101   return this.tree.inorderTraversal(q);
102 };
